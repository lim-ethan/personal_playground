# 3. Stream에 적재된 Record 읽기

```python
import boto3
import time
stream_name = 'stream_based_on_code'
region_name = 'ap-northeast-2'
client = boto3.client(
    'kinesis',
    region_name=region_name,
    aws_access_key_id='your-access-key',
    aws_secret_access_key='your-secret-key'
)

response = client.describe_stream(StreamName=stream_name)
shard_id = response['StreamDescription']['Shards'][0]['ShardId']

shard_iterator = client.get_shard_iterator(
    StreamName=stream_name,
    ShardId=shard_id,
    ShardIteratorType='TRIM_HORIZON'  # 처음부터 읽기. 이후에는 'LATEST'를 사용해도 됨
)['ShardIterator']

while True:
    response = client.get_records(ShardIterator=shard_iterator, Limit=10)
    records = response['Records']
    for record in records:
        print(record['Data'].decode('utf-8'))
    shard_iterator = response['NextShardIterator']
    time.sleep(0.3)
```
